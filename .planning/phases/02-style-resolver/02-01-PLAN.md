    ```typescript
    import { prisma } from '@/lib/prisma'
    import { ART_STYLES } from '@/lib/constants'
    import { isUserStyle, extractUserStyleId } from './style-namespace'

    /**
     * 异步解析风格提示词
     * - 系统预设：从 ART_STYLES 常量实时查找（userId 不参与）
     * - 用户自定义：从 UserStyle 表查询（userId 必须参与，防止跨用户访问）
     * - 找不到时返回 null，调用方通过 ?? '' 保持向后兼容
     *
     * @param artStyle - 风格标识符，如 'american-comic' 或 'user:uuid'
     * @param userId - 当前用户 ID（用户自定义风格查询时必须）
     * @param locale - 语言：'zh' | 'en'
     * @returns 对应风格的提示词字符串，或 null（找不到时）
     */
    export async function resolveStylePrompt(
      artStyle: string,
      userId: string,
      locale: 'zh' | 'en',
    ): Promise<string | null> {
      // 系统预设路径：直接从常量查找，不查库
      if (!isUserStyle(artStyle)) {
        const style = ART_STYLES.find((s) => s.value === artStyle)
        if (!style) return null
        return locale === 'en' ? style.promptEn : style.promptZh
      }

      // 用户自定义路径：从 DB 查询，userId 必须参与（D-10 安全要求）
      const styleId = extractUserStyleId(artStyle)
      const userStyle = await prisma.userStyle.findUnique({
        where: {
          id: styleId,
          userId, // 防止用户 A 访问用户 B 的风格
        },
        select: {
          promptZh: true,
          promptEn: true,
        },
      })

      if (!userStyle) return null
      return locale === 'en' ? userStyle.promptEn : userStyle.promptZh
    }
    ```

    注意事项（per D-10, D-12, D-14, D-15）：
    - 函数名：`resolveStylePrompt`（不是 `getStylePrompt`，per D-14）
    - 返回类型：`Promise<string | null>`（不是 `UserStylePrompt`，per D-10）
    - userId 必须在 DB 查询的 where 条件中（安全性 per D-10）
    - 返回 null 而非空字符串（per D-12）
    - 预设风格路径不查库（per D-10 + research Pitfall 3）
    - 不做数量检查（per D-13：Phase 5 才做）
    - 不做缓存（deferred，按需 Phase 4 评估）
