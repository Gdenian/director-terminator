3. **修改验证逻辑**：
   ```typescript
   async function validateArtStyleField(
     value: unknown,
     userId: string,
   ): Promise<string> {
     if (typeof value !== 'string') {
       throw new ApiError('INVALID_PARAMS', {
         code: 'INVALID_ART_STYLE',
         field: 'artStyle',
         message: 'artStyle must be a supported value',
       })
     }
     const artStyle = value.trim()

     // 系统预设路径
     if (!isUserStyle(artStyle)) {
       if (!isArtStyleValue(artStyle)) {
         throw new ApiError('INVALID_PARAMS', {
           code: 'INVALID_ART_STYLE',
           field: 'artStyle',
           message: 'artStyle must be a supported value',
         })
       }
       return artStyle
     }

     // 用户风格路径：验证存在性和所有权
     const styleId = extractUserStyleId(artStyle)
     const userStyle = await prisma.userStyle.findUnique({
       where: { id: styleId, userId },
       select: { id: true },
     })
     if (!userStyle) {
       throw new ApiError('INVALID_PARAMS', {
         code: 'INVALID_ART_STYLE',
         field: 'artStyle',
         message: '指定的自定义风格不存在',
       })
     }
     return artStyle
   }
   ```

4. **更新调用点**（在 PATCH handler 中，约第 313 行）：

   ```typescript
   // 之前
   updateData[field] = validateArtStyleField(body[field])

   // 之后
   updateData[field] = await validateArtStyleField(body[field], session.user.id)
   ```

**关键点：**

- 查询时 userId 必须在 where 条件中，防止跨用户访问
- 保持原有错误码 INVALID_ART_STYLE 的兼容性
